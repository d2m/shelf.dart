ShelfRequest class

Represents an HTTP request to be processed by a Shelf application.

Properties are inspired by those in the Rack env parameter.

Rack Spec

Rack was inspired by PEP 3333.

class ShelfRequest extends UnmodifiableMapView<String, String> {

 static const String CONTENT_LENGTH = 'CONTENT_LENGTH';
 static const String CONTENT_TYPE = 'CONTENT_TYPE';
 static const String PATH_INFO = 'PATH_INFO';
 static const String QUERY_STRING = 'QUERY_STRING';
 static const String REQUEST_METHOD = 'REQUEST_METHOD';

 /**
  * The initial portion of the request URL's "path" that corresponds to the
  * application object, so that the application knows its virtual "location".
  * This may be an empty string, if the application corresponds to the "root"
  * of the server.
  */
 static const String SCRIPT_NAME = 'SCRIPT_NAME';

 static const String SERVER_NAME = 'SERVER_NAME';
 static const String SERVER_PORT = 'SERVER_PORT';
 static const String SERVER_PROTOCOL = 'SERVER_PROTOCOL';

 //
 // Core Values
 //

 /**
  * The contents of any Content-Length fields in the HTTP request. May be
  * [:null:].
  *
  * Represented by the key [CONTENT_LENGTH].
  */
 int get contentLength => _intOrNull(this[CONTENT_LENGTH]);

 /**
  * The contents of any Content-Type fields in the HTTP request. May be
  * [:null:].
  *
  * Represented by the key [CONTENT_TYPE].
  */
 String get contentType => this[CONTENT_TYPE];

 /**
  * The remainder of the request URL's "path", designating the virtual
  * "location" of the request's target within the application. This may be an
  * empty string, if the request URL targets the application root and does not
  * have a trailing slash.
  *
  * Represented by the key [PATH_INFO].
  */
 String get pathInfo => this[PATH_INFO];

 /**
  * The portion of the request URL that follows the "?", if any. May be `null`.
  *
  * Represented by the key [QUERY_STRING].
  */
 String get queryString => this[QUERY_STRING];

 /**
  * The HTTP request method, such as "GET" or "POST". This cannot ever be an
  * empty string, and so is always required.
  *
  * Represented by the key [REQUEST_METHOD].
  */
 String get requestMethod => this[REQUEST_METHOD];

 String get scriptName => this[SCRIPT_NAME];

 /**
  * Represented by the key [SERVER_PORT].
  */
 int get serverPort => _intOrNull(this[SERVER_PORT]);

 /**
  * The version of the protocol the client used to send the request. Typically
  * this will be something like "HTTP/1.0" or "HTTP/1.1" and may be used by the
  * application to determine how to treat any HTTP request headers.
  *
  * Represented by the key [SERVER_PROTOCOL].
  *
  * (This variable should probably be called REQUEST_PROTOCOL, since it denotes
  * the protocol used in the request, and is not necessarily the protocol that
  * will be used in the server's response. However, for compatibility with CGI
  * we have to keep the existing name.)
  */
 String get serverProtocol => this[SERVER_PROTOCOL];

 // TODO(kevmoo): validate that required items are present and in the correct
 //               format
 ShelfRequest(Map<String, String> headers) :
   super(new SplayTreeMap.from(headers)) {

   assert(requestMethod.isNotEmpty);
   assert(requestMethod == requestMethod.toUpperCase());

   // The <tt>SCRIPT_NAME</tt>, if non-empty, must start with <tt>/</tt>
   assert(scriptName != null);
   assert(scriptName.isEmpty || scriptName.startsWith('/'));

   // The <tt>PATH_INFO</tt>, if non-empty, must start with <tt>/</tt>
   assert(pathInfo != null);
   assert(pathInfo.isEmpty || pathInfo.startsWith('/'));

   // The <tt>CONTENT_LENGTH</tt>, if given, must consist of digits only.
   assert(contentLength == null || contentLength >= 0);



   /**
* One of <tt>SCRIPT_NAME</tt> or <tt>PATH_INFO</tt> must be
 set.  <tt>PATH_INFO</tt> should be <tt>/</tt> if
 <tt>SCRIPT_NAME</tt> is empty.
 <tt>SCRIPT_NAME</tt> never should be <tt>/</tt>, but instead be empty.
    */
 }

 /*
  * REMOTE_ADDR 127.0.0.1
  * REMOTE_HOST localhost
  * REQUEST_METHOD  GET
  * REQUEST_PATH  /dancingfrog/cool
  * REQUEST_URI http://localhost:9292/dancingfrog/cool?foo=bar
  * SCRIPT_NAME
  * SERVER_NAME localhost
  */

 //
 // Other
 //

 /**
  * Convenience property to access [pathInfo] data as a [List].
  *
  * No corresponding key.
  */
 List<String> get pathSegments {
   var segs = pathInfo.split('/');
   if (segs.length > 0 && segs.first.isEmpty) {
     segs.removeAt(0);
   }
   if (segs.last.isEmpty) {
     segs.removeLast();
   }
   assert(segs.every((s) => !s.isEmpty));
   return segs;
 }

 static int _intOrNull(String input) =>
     input == null ? null : int.parse(input);
}
