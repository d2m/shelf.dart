MiddlewareBuilder abstract class

An abstract class for creating ShelfMiddleware instances by overriding functions.

abstract class MiddlewareBuilder {

 const MiddlewareBuilder();

 /**
  * **SHOULD NOT BE CALLED DIRECTLY.**
  *
  * Override in a subclass to process [request] before the inner handler.
  *
  * Return [:null:] to pass the [request] to the inner handler.
  *
  * The default implementation returns [:null:].
  */
 dynamic handleRequest(ShelfRequest request) => null;

 /**
  * **SHOULD NOT BE CALLED DIRECTLY.**
  *
  * Override in a subclass to process [request] after the inner handler.
  *
  * [response] is the retutrn value from the inner handler.
  *
  * The return value must be an instance of `ShelfResponse` or
  * `Future<ShelfResponse>`.
  *
  * The default implementation returns [response].
  */
 dynamic processResponse(ShelfRequest request, ShelfResponse response) =>
     response;

 ShelfMiddleware get middleware => getHandler;

 ShelfHandler getHandler(ShelfHandler innerHandler) {
   return (ShelfRequest request) {
     return new Future.sync(() => handleRequest(request))
       .then((ShelfResponse response) {
         if(response != null) return response;
         return innerHandler(request);
       })
       .then((ShelfResponse response) {
         return processResponse(request, response);
       });
   };
 }
}
