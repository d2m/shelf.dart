ShelfStack class 

class ShelfStack {
 final ShelfStack _parent;
 final ShelfMiddleware _builder;

 const ShelfStack()
     : _builder = null,
       _parent = null;

 ShelfStack._(this._builder, this._parent) {
   assert(_builder != null);
   assert(_parent != null);
 }

 ShelfStack addMiddleware(ShelfMiddleware builder) =>
     new ShelfStack._(builder, this);

 ShelfHandler addApp(ShelfHandler handler) {
   assert(handler != null);

   if(_builder == null) {
     assert(_parent == null);
     return handler;
   } else {
     return _parent.addApp(_builder(handler));
   }
 }
}
